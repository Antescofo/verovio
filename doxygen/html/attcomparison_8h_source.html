58 class AttDurExtreme : public AttComparison {
59 
60 public:
61  AttDurExtreme(DurExtreme extremeType) : AttComparison(OBJECT)
62  {
63  m_extremeType = extremeType;
64  if (m_extremeType == LONGEST)
65  m_extremeDur = -VRV_UNSET;
66  else
67  m_extremeDur = VRV_UNSET;
68  }
69 
70  virtual bool operator()(Object *object)
71  {
72  if (!object->HasInterface(INTERFACE_DURATION)) return false;
73  DurationInterface *interface = dynamic_cast<DurationInterface *>(object);
74  assert(interface);
75  if (interface->HasDur()) {
76  if ((m_extremeType == LONGEST) && (interface->GetActualDur() < m_extremeDur)) {
77  m_extremeDur = interface->GetActualDur();
78  return true;
79  }
80  else if ((m_extremeType == SHORTEST) && (interface->GetActualDur() > m_extremeDur)) {
81  m_extremeDur = interface->GetActualDur();
82  return true;
83  }
84  }
85  return false;
86  }
87 
88 private:
89  int m_extremeDur;
90  DurExtreme m_extremeType;
91 };
